  3. Deploy → New deployment → type: Web app
  4. Execute as: Me · Who has access: Anyone
  5. Copy the Web app URL, paste it below on both phones
function doGet(e) { var d = PropertiesService.getScriptProperties() .getProperty('data') || '[]'; return ContentService.createTextOutput(d) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { PropertiesService.getScriptProperties() .setProperty('data', e.postData.contents); return ContentService.createTextOutput('ok'); }
